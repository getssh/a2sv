#  1. INTEGER n
#  2. INTEGER_ARRAY arr

def insertionSort1(n, arr):
    # Write your code here
    for i in range(n):
        cur_val = arr[i]
        idx = i - 1
        while (idx >= 0 and cur_val < arr[idx]):
            arr[idx + 1] = arr[idx]
            print(" ".join(map(str, arr)))
            idx -= 1
            
        arr[idx+1] = cur_val
    
    print(" ".join(map(str, arr)))

if __name__ == '__main__':
    n = int(input().strip())

    arr = list(map(int, input().rstrip().split()))

    insertionSort1(n, arr)
